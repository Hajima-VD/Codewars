      function breakCamelCase(str) {

      }
Cортировка массива по возрастанию . Для этого можно использовать метод sort(): В этом примере мы передаем функцию сравнения в метод sort(), которая сравнивает два элемента массива a и b. Если результат вызова этой функции отрицательный, то a будет расположен перед b. Если результат положительный, то b будет перед a. 
                     

      let myArray = [3, 1, 4, 2, 5];
      myArray.sort(function(a, b) {
        return a - b;
      });
      
      console.log(myArray); // [1, 2, 3, 4, 5]

Код который возводит каждый элемент массива во 2 степень на JavaScript. 
        
  const arr = [1, 2, 3, 4, 5];
  const squaredArr = arr.map(num => {
    return num ** 2;
  });
  console.log(squaredArr); // [1, 4, 9, 16, 25]
Этот код создаёт новый массив, в котором каждый элемент возводится во 2 степень. Метод map() позволяет применять функцию ко всем элементам массива и возвращать новый массив с изменёнными значениями. 
          
  function squareDigits(num) {
    let digits = num.toString().split(''); // превращаем число в массив цифр
    let squaredDigits = digits.map(n => n * n); // возводим каждую цифру в квадрат
    let result = squaredDigits.join(''); // склеиваем значения в новое число
    return parseInt(result); // превращаем строку в число и возвращаем результат
  }
