Build a function that returns an array of integers from n to 1 where n>0. Example : n=5 --> [5,4,3,2,1] 
                     
                        const reverseSeq = n => {
                            // объявляем пустой массив
                           let myArray = [];
                           
                           // заполняем массив значениями от 0 до 4
                           for (let i = 1; i <= n; i++) {
                             myArray.push(i);
                           }
                           return myArray.reverse();
                             };
